üíª Python API Usage

Use the system programmatically in your Python projects:

from financial_modelling import DataManager, PortfolioOptimizer # Fetch data dm = DataManager() dm.fetch_data(['VAS.AX', 'VGS.AX', 'VDHG.AX']) # Get returns returns = dm.get_returns() # Optimize portfolio optimizer = PortfolioOptimizer(returns) result = optimizer.optimize_max_sharpe() # View results print(f"Weights: {result.weights}") print(f"Expected Return: {result.expected_return:.2%}") print(f"Volatility: {result.volatility:.2%}") print(f"Sharpe Ratio: {result.sharpe_ratio:.2f}")
